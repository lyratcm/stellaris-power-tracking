clear_situation_true_omni = {
    every_situation = {
        limit = {
            is_situation_type = delta_to_omni
			has_situation_flag = true_tech
        }
        destroy_situation = this
    }
	if = {
		limit = {
			is_variable_set = delta_to_$TYPE$_progress
		}
		clear_variable = delta_to_$TYPE$_progress
	}
	if = {
		limit = {
			has_country_flag = $TYPE$
		}
		remove_country_flag = $TYPE$
	}

}
